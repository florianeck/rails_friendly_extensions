def self.year_letter(year=Time.now.year, letters = LETTERS)
  
  steps = []
  
  d = (year - START_YEAR)
  n = d.fdiv(letters.size).ceil
  string = ""
  
  n.times do |i| 
    # First 
    if i == 0
      string << letters[d%letters.size]
    else
      x = d.fdiv(letters.size*(i+1)).round
      string << letters[x]
    end    
  end  
  
  return string
end
